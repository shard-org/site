Bubble Sort

The defined script will be called on sharc run.
We're also using the lib tag to not have to directly edit linker arguments.

:name bubble_sort
:arch x86_64 linux
:lib c

:verb run /bin/sh {
   sharc #FILE
   chmod +x #NAME
   ./#NAME
}

entry:
   %array [2:] = {2, 8, 9, 7, 4, 3, 6, 5, 1, 0}

   !bubble_sort array, 10

   // print the array
   loop {'i ++}, (i = 10), <= {%i 2} {
      $printf "%d\n", array.i
   }

   ret

bubble_sort [2:] array, 2 len:
   loop {'i ++}, (i < len), <= {%i 2}:
      loop {'j ++}, (j < len-i-1) <= {%j 2}:
         (array.j > [array.j + 1]):
            'array.j : [array.j + 1]
   ret
