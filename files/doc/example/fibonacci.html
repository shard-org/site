Fibonacci

Here we're telling the linker to link libc.
:arch x86_64 linux
:linker #LINKER -lc

fibonacci 2 n, -> 2 {
   (n <= 1): ret n
   ret !fibonacci (n - 1) + !fibonacci (n - 2)
}

entry:
   %terms 2
   $scanf "%d", terms

   %i 2
   loop, ('i ++ & i < terms):
      $printf "%d\n", (!fibonacci i)

   ret
